/**
 * Web3 contract instance for interacting with the deployed contract.
 * @type {Web3.eth.Contract}
 */
const contract = new web3.eth.Contract(contractABI, contractAddress);

/**
 * Wallet private key for the deployer account.
 * This key should be kept private and secure.
 * @type {string}
 */
const PRIVATE_KEY = "0x214b606ee0c2e0bfb0e601424ee9a9f3e753e43ed14996d2da05f59c66eebaa5"; // Truffle local private key 

/**
 * Convert the private key to a Web3 account object.
 * @type {Web3.eth.Account}
 */
const deployerAccount = web3.eth.accounts.privateKeyToAccount(PRIVATE_KEY);

// Add the deployer's account to the Web3 wallet
web3.eth.accounts.wallet.add(deployerAccount);

/**
 * Set the default account to the deployer's address.
 * This will be used to sign transactions.
 */
web3.eth.defaultAccount = deployerAccount.address;

/**
 * Get the current contract owner.
 * @async
 * @returns {Promise<string>} The address of the contract owner.
 * @throws {Error} If an error occurs while fetching the contract owner.
 */
async function getContractOwner() {
  try {
    const ownerAddress = await contract.methods.owner().call();
    console.log("Contract Owner:", ownerAddress);
    return ownerAddress;
  } catch (error) {
    console.error("Error fetching contract owner:", error);
    throw error; // Re-throw the error for further handling
  }
}
